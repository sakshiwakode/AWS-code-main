     $ sudo yum update -y \n
-------> for sql client
     $ sudo yum install mysql -y \n

$ sudo yum install python3 -y \n

 $ sudo python3 -m pip install boto3 \n
 $ sudo python3 -m pip install flask \n
-------> for running application
     $ sudo python3 EmpApp.py \n
